## Icons

You can use the `icon` and `iconProps` props on the `Dropdown` component to style the button icon.

    <Icon name="cat" color="dark" mr="15px" />
    <Text inline>Cat</Text>
  </Dropdown.Option>

## Advanced usage

<ItunesSearch>
  {({search, results, onChange}) => (
    <Dropdown
      icon="search"
      height="tall"
      containerProps={{maxHeight: "150px"}}
      label={(
        <ItunesSearch.Input
          value={search}
          onChange={(e) => onChange(e.target.value)} />
      )}>
      {results.length ? (
        results.map(({collectionId, artworkUrl60, artistName, collectionName}) => (
          <Dropdown.Option key={collectionId} value={collectionId}>
            <View mr="15px">
              <img src={artworkUrl60} height="30px" width="30px" />
            </View>
            {artistName} - {collectionName}
          </Dropdown.Option>
        ))
      ) : search.length ? (
        <View mr="15px" ml="15px">
          <Text>No results for "{search}"</Text>
        </View>
      ) : (
        <View mr="15px" ml="15px">
          <Text>Start typing to search albums</Text>
        </View>
      )}
    </Dropdown>
  )}
</ItunesSearch>
