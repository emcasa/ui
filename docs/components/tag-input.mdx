## With search input

<Playground style={{minHeight: "250px"}}>
  <TagSearch tags={tags}>
    {({loading, search, results, onChange}) =>
      <TagInput
        placeholder="Search tags"
        options={results && results.length > 0 ? results : tags}
        onChangeText={onChange}
        >
        {loading ? (
          <View mr="15px" ml="15px">
            <Text>Loading...</Text>
          </View>
        ) : results.length ? (
          undefined
        ) : search.length ? (
          <View mr="15px" ml="15px">
            <Text>No results for "{search}"</Text>
          </View>
        ) : (
          <View mr="15px" ml="15px">
            <Text>Start typing to search tags</Text>
          </View>
        )}
      </TagInput>
    }
  </TagSearch>
</Playground>
